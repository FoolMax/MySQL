创建库

```sql
-- utf-8库，collate utf8_general_ci不区分大小写
CREATE DATABASE 库名 DEFAULT CHARSET utf8 COLLATE utf8_general_ci;
 
-- gbk库
CREATE DATABASE 库名 DEFAULT CHARACTER SET gbk COLLATE gbk_chinese_ci;
```

查看库

```sql
-- 查看所有库
show databases;
```

删除库

```sql
drop database 库名;
```

进入库

```sql
use 库名;
```
